104 template <stream::state_type S, typename T>
105 inline parser<S, std::vector<T>>
106 count(uint32_t n, parser<S, T> p) {
107  using Replicate =
108  std::function<parser<S, std::vector<T>>(uint32_t, std::vector<T>)>;
109 
110  if (n == 0) {
111  return pure<S>({});
112  } else {
113  Replicate replicate;
114  replicate = [=](uint32_t n, std::vector<T> &&acc) {
115  if (n == 0) {
116  return pure<S>(std::move(acc));
117  } else {
118  return p >>= [&replicate, &acc, n](T v) {
119  acc.push_back(std::move(v));
120  return replicate(n - 1, acc);
121  };
122  }
123  };
124  return replicate(n, {});
125  }
126 }

585 inline auto str = [](std::string s) -> parser<string_state, std::string> {
586  if (s == "") {

588  }
589  std::vector<parser<string_state, char>> chparsers;
