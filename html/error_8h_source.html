73  : position(position)

76  // if the messages is empty
77  bool empty() { return messages.empty(); }
78  bool is_unknown_error() { return empty(); }
79 
80  // merge two errors to get a new error.
81  friend parser_error operator+(const parser_error &e1,
82  const parser_error &e2) {
83  if (!e1.messages.empty() && e2.messages.empty()) {
84  return e2;
85  } else if (e1.messages.empty() && !e2.messages.empty()) {
86  return e2;
87  }
88 
89  if (e1.position == e2.position) {
90  Messages msg{};
91  msg.insert(msg.end(), e1.messages.begin(), e1.messages.end());
92  msg.insert(msg.end(), e2.messages.begin(), e2.messages.end());
93 
94  return parser_error{ e1.position, msg };
95  } else if (e1.position > e2.position) {
96  return e1;
97  } else {
98  return e2;
99  }
100  }

120  ss << "line: " << position.line << ", column: " << position.col << ". ";
121  if (messages.size() == 0) {
122  ss << "no error info";
123  } else {
124  ss << messages.size() << " errors occurred.\n";
125  ss << "details (lastest on top):\n";
