56 // The return type of a parser. It contains the current state (stream) and
57 // parsed result. `consumed` and `ok` are used to indicate the state of the
58 // parser.
60 template <stream::state_type S, typename T>
61 class reply {
62  // NOTE: This will be the value get passed to
63  // the next continuation in ok case.
64  public:
65  using type = T;
66  using stream = T;
67  bool consumed;
68  bool ok;
69  std::optional<T> value;
70  S state;
71 
72  parser_error error;
73 
74  reply()
75  : consumed(false)
76  , ok(false)
77  , value(std::nullopt)
78  , state()
79  , error() {}
80 
97  reply(bool consumed, bool ok, std::optional<T> value, S state,
98  parser_error error)
99  : consumed(consumed)
100  , ok(ok)
101  , value(value)
102  , state(state)
103  , error(error) {
104  assert(ok ? value.has_value() : !value.has_value());
105  }
106 
109  T get() {
110  if (value.has_value()) {
111  return value.value();
112  } else {
113  throw parse_failed(error.to_string());
114  }
115  }
116 
118  const reply &merge_with_error(const parser_error &new_error) {
119  error = error + new_error;
120  std::cerr << error.to_string() << std::endl;
121  return *this;
122  }

126  template <typename Fn,
127  typename U = typename function_traits<Fn>::return_type>
128  constexpr reply<S, U> map(Fn fn) const {
129  if (ok) {
130  return { consumed, ok, std::optional{ fn(value.value()) }, state,
131  error };
132  } else {
133  return { consumed, ok, std::nullopt, state, error };
134  }
135  }
136 
137  // make some smart constructors to avoid invalid state. If we only create
138  // Reply with these functions, we will never have invalid Reply. e.g an
139  // empty reply with non std::nullopt value.
140 
142  static reply<S, T> mk_consumed_ok_reply(T value, S state,
143  parser_error error) {
144  return { true, true, { value }, state, error };
145  }
146 
148  static reply<S, T> mk_consumed_err_reply(S state, parser_error error) {
149  return { true, false, {}, state, error };
150  }
151 
153  static reply<S, T> mk_empty_ok_reply(T value, S state, parser_error error) {
154  return { false, true, { value }, state, error };
155  }
156 
158  static reply<S, T> mk_empty_err_reply(S state, parser_error error) {
159  return { false, false, {}, state, error };
160  }
161 };

170 // We have four different continuations for four possible parser states, each
171 // callback corresponding to one specific state of the reply.
172 //
173 // 1. consumed_ok: The stream was consumed and the last parser was succeed.
174 // 2. consumed_err: The stream was consumed but the last parser was failed.
175 // 3. empty_ok: The stream was not consumed but it's consided as a success.
176 // 4. empty_err: The stream was not consumed and it's considered as a failure.
177 //
178 // This is how we handle errors and back traces.
179 template <stream::state_type S, typename T>
180 struct Conts {
181  OkContinuation<S, T> consumed_ok;
182  ErrContinuation consumed_err;
183  OkContinuation<S, T> empty_ok;
184  ErrContinuation empty_err;
185 };

252  // Using shared_ptr here so we can ensure the life time of unparser last
253  // longer than the wrapper type.
254  //
255  // Often times we want to define a standalone parser p1, compose it with
256  // othe parsers by some combinators to create a new parser p2, and use both
257  // p1 and p2. The lifetime of this scheme can be tricky to handle.
258  //
259  // 1. It will be very wasteful if we copy both p1 to p2.
260  // 2. If we pass by reference we will lose intermediate parser in a long
261  // expression. e.g p = p1.map(f1).map(f2); the parser created by
262  // p1.map(f1) lives till the end of the statement.
263  // 3. If we move p1 into p2 we can't use it as a
264  // standalone parser anymore. shared_ptr helps
265  // solve problems above. The whole parser is
266  // just a wrapper over the shared_ptr, copy
267  // Parser will copy 16 bytes.
268  //
269  // Continuation and shared ptr unparser are
270  // implementation details, and you should never
271  // need to interact with them direclty.
272  std::shared_ptr<parser_fn<S, T>> unparser;
273 
274  parser(const parser_fn<S, T> &parse)
275  : unparser(std::make_shared<parser_fn<S, T>>(parse)) {}
276 
277  parser(parser_fn<S, T> &&parse)
278  : unparser(std::make_shared<parser_fn<S, T>>(std::move(parse))) {}

305  // create parser from a transition function. This is a low level helper, you
306  // should never have the need to call this function.
307  static parser<S, T> create(std::function<reply<S, T>(S)> transition) {
308  return parser([transition](S state, Conts<S, T> cont) {
309  reply<S, T> r = transition(state);
310 
311  assert((r.ok && r.value != std::nullopt) ||
312  (!r.ok && r.value == std::nullopt));
313 
314  if (r.consumed) {
315  r.ok
316 
317  ? cont.consumed_ok(r)
318  : cont.consumed_err(r.error);
319  } else {
320  r.ok
321 
322  ? cont.empty_ok(r)
323  : cont.empty_err(r.error);
324  }
325  return r.ok;
326  });
327  }

374 template <stream::state_type S, typename T>
375 reply<S, T>
376 parser<S, T>::operator()(const S &state) {
377  reply<S, T> r;
378 
379  // The entrance callback. This function will be the bottom of a call to a
380  // parser, and it simply capture and return the current reply.
381 
382  auto ok = [&r](auto rep) -> bool {
383  r = rep;
384  return r.ok;
385  };
386 
387  (*unparser)( // run the ParserFn
388 
389  state,
390 
391  {
392 
393  .consumed_ok = ok,
394 
395  // handle consumed but is error.
396  .consumed_err = [&r, &state](parser_error err) -> bool {
397  r = reply<S, T>::mk_consumed_err_reply(state, err);
398  return r.ok;
399  },
400  .empty_ok = ok,
401 
402  // handle not consumed and is error.
403  .empty_err = [&r, &state](parser_error err) -> bool {
404  r = reply<S, T>::mk_empty_err_reply(state, err);
405  return r.ok;
406  }
407 
408  });
409  return r;
410 }

414 template <stream::state_type S, typename T>
415 template <typename Fn, typename U>
416 parser<S, U>
417 parser<S, T>::map(const Fn &fn) {
418  static_assert(std::is_convertible_v<Fn, std::function<U(T)>>,
419  "Function to map has the wrong type");
420 
421  return parser<S, U>(
422 
423  [&fn, p = unparser](S state, Conts<S, U> cont) {
424  // map on continuation.
425  auto mapped_ok = [&cont, &fn](reply<S, T> rep) -> bool {
426  return cont.consumed_ok(rep.map(fn));
427  };
428 
429  return (*p)(
430 
431  state,
432 
433  // only need to map continuations that
434  // carry values.
435  {
436 
437  .consumed_ok = mapped_ok,
438  .consumed_err = cont.consumed_err,
439  .empty_ok = mapped_ok,
440  .empty_err = cont.empty_err
441 
442  });
443  });

448 template <stream::state_type S, typename T>
449 template <typename Fm, typename U>
450 parser<S, U>
451 parser<S, T>::bind(Fm fm) {
452  static_assert(std::is_convertible_v<Fm, std::function<parser<S, U>(T)>>,
453  "Monadic function for bind has the wrong "
454  "type");
455 
456  return parser<S, U>([=, p = unparser](S state, Conts<S, U> cont) {
457  auto consumer_ok = [=](reply<S, T> rep) -> bool {
458  assert(rep.value.has_value());
459 
460  // for consumed case just go through.
461  auto consumed_ok = cont.consumed_ok;
462  auto consumed_err = cont.consumed_err;
463 
464  // go with the consumed continuation when m.unparser doesn't consume
465  // but ok.
466  auto pempty_ok = [=](reply<S, U> rep1) -> bool {
467  rep1.error = rep.error + rep1.error;
468  return cont.consumed_ok(rep1);
469  };
470 
471  auto pempty_err = [=](parser_error e) -> bool {
472  parser_error error = rep.error + e;
473  return cont.consumed_err(error);
474  };
475 
476  // call parser.
477  parser<S, U> m = fm(rep.value.value());
478  return (*m.unparser)(
479 
480  rep.state,
481 
482  {
483 
484  .consumed_ok = consumed_ok,
485  .consumed_err = consumed_err,
486  .empty_ok = pempty_ok,
487  .empty_err = pempty_err
488 
489  });
490  };
491 
492  // doing the similar thing for empty ok.
493  auto empty_ok = [=](reply<S, T> rep) {
494  assert(rep.value.has_value());
495 
496  auto consumed_ok = cont.consumed_ok;
497  auto consumed_err = cont.consumed_err;
498 
499  auto pempty_ok = [=](reply<S, U> rep1) -> bool {
500  rep1.error = rep.error + rep1.error;
501  return cont.empty_ok(rep1);
502  };
503 
504  auto pempty_err = [=](parser_error e) -> bool {
505  parser_error error = rep.error + e;
506  return cont.empty_err(e);
507  };
508 
509  parser<S, U> m = fm(rep.value.value());
510  return (*m.unparser)(
511 
512  rep.state,
513 
514  {
515 
516  .consumed_ok = consumed_ok,
517  .consumed_err = consumed_err,
518  .empty_ok = pempty_ok,
519  .empty_err = pempty_err
520 
521  });
522  };
523 
524  return (*p)(state,
525 
526  {
527 
528  .consumed_ok = consumer_ok,
529  .consumed_err = cont.consumed_err,
530  .empty_ok = empty_ok,
531  .empty_err = cont.empty_err
532 
533  });
534  });
535 }

582 te !mplate<stream::state_type S, typename T> parser<S, T> parser<S, T>::alt(
583  parser<S, T> n) {
584  return parser([=, m = unparser](S state, Conts<S, T> cont) {
585  // when m fails without consuming anything,
586  // parse n.
587  auto mempty_err = [=](parser_error error) -> bool {
588  auto nempty_ok = [=](reply<S, T> rep) {
589  rep.error = rep.error + error;
590  return cont.empty_ok(rep);
591  };
592 
593  // if nempty is also failed, return an
594  // error. this case is actually the identify
595  // of `alt` operator.
596  auto nempty_err = [=](parser_error error1) -> bool {
597  return cont.empty_err(error + error1);
598  };
599 
600  return (*n.unparser)(
601 
602  state,
603 
604  {
605 
606  .consumed_ok = cont.consumed_ok,
607  .consumed_err = cont.consumed_err,
608  .empty_ok = nempty_ok,
609  .empty_err = nempty_err
610 
611  });
612  };
613 
614  return (*m)(
615 
616  state,
617 
618  {
619 
620  .consumed_ok = cont.consumed_ok,
621  .consumed_err = cont.consumed_err,
622  .empty_ok = cont.empty_ok,
623  .empty_err = mempty_err
624 
625  });
626  });
627 }

717 template <stream::state_type S, typename T, typename AccumFn>
718 parser<S, std::vector<T>>
719 many_accum(AccumFn fn, parser<S, T> p) {
720  static_assert(
721  std::is_convertible_v<AccumFn,
722  std::function<std::vector<T>(T, std::vector<T>)>>,
723  "AccumFn has the wrong type");
724 
725  return parser<S, std::vector<T>>([=](S state,
726  Conts<S, std::vector<T>> cont) {
727  // TODO: Now all vector are copied. handle
728  // this later. recursively accumulate result
729  // from p
730 
731  // The recursive function to accumulate the parsing result.
732  std::function<bool(std::vector<T> acc, reply<S, T> rep)> walk;
733  walk = [&walk, &fn, &cont, p](std::vector<T> acc, reply<S, T> rep) {
734  return (*p.unparser)(
735  rep.state,
736 
737  {
738 
739  .consumed_ok = [&walk, &fn, acc](reply<S, T> rep) -> bool {
740  auto v = rep.value.value();
741  walk(fn(v, acc), rep);
742  return rep.ok;
743  },
744 
745  .consumed_err = cont.consumed_err,
746 
747  // this case should never happen. You can't have a parser
748  // accepts empty string keeps running.
749  // It's questionable to use an exception here. Maybe abort
750  // the program?
751  .empty_ok =
752  [](reply<S, T> rep) {
753  throw bad_many_combinator();
754  return rep.ok;
755  },
756 
757  // the base case. once we hit this case, treat it as
758  // success and exit the parser.
759  .empty_err =
760  [=]([[maybe_unused]] parser_error error) -> bool {
761  auto rep1 =
762  reply<S, std::vector<T>>::mk_consumed_ok_reply(
763  acc, rep.state, rep.error);
764 
765  return cont.consumed_ok(rep1);
766  }
767 
768  });
769  };
770 
771  return (*p.unparser)(
772 
773  state,
774 
775  {
776 
777  // if consumed ok, invoke walk, which will call walk once again
778  // on success.
779  .consumed_ok = [&walk](reply<S, T> rep) -> bool {
780  auto v = rep.get();
781  walk({ v }, rep);
782  return rep.ok;
783  },
784 
785  .consumed_err = cont.consumed_err,
786 
787  .empty_ok = [](reply<S, T> rep) -> bool {
788  throw bad_many_combinator();
789  return rep.ok;
790  },
791 
792  .empty_err = [&cont, state](parser_error error) -> bool {
793  auto rep = reply<S, std::vector<T>>::mk_empty_ok_reply(
794  { {} }, state, error);
795  return cont.empty_ok(rep);
796  }
797 
798  });
799  });
800 }

829 template <stream::state_type S, typename T, typename PrettyPrint,
830  typename Match>
831 parser<S, T>
832 token(PrettyPrint pretty_print, Match match) {
833  using V = typename S::ValueType;
834  using D = typename S::StreamType;
835 
836  // some constraints
837  static_assert(
838  std::is_convertible_v<PrettyPrint, std::function<std::string(V)>>,
839  "pretty printer with wrong type");
840 
841  static_assert(std::is_convertible_v<Match, std::function<bool(V)>>,
842  "match has the wrong type");
843 
844  return parser<S, T>(
845 
846  [match, pretty_print](S state, Conts<S, T> cont) {
847  // fetch from stream.
848  std::optional<std::tuple<V, D>> r = state.uncons();
849 
850  if (!r.has_value()) {
851  auto pos = state.get_position();
852  auto error = unexpect_error(pos, "The stream is empty");
853  return cont.empty_err(error);
854  }
855 
856  auto [v, stream] = r.value(); // peek
857  if (match(v)) {
858  // valid token, construct a new reply
859  // with the token as it's value.
860  auto next_state = state.eat(state.next_position());
861  state = std::move(next_state);
862  auto err = unknown_error(state);
863  auto rep = reply<S, T>::mk_consumed_ok_reply({ v }, state, err);
864 
865  rep.value = { v };
866  rep.state = state;
867  rep.error = rep.error + unknown_error(state);
868 
869  return cont.consumed_ok(rep);
870  } else {
871  auto pos = state.get_position();
872  auto msg = pretty_print(v);
873  auto error = unexpect_error(pos, msg);
874  return cont.empty_err(error);
875  }
876  });
877 }
